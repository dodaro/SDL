@entity Clause: id: int;
@entity Vars: id: int;
@entity In_clause: clause: int, literal: int;
@entity Assignment: variable: Vars;

guess from Vars as v
	at least 1 and at most 2
		Assignment 
			where Assignment.variable=v;

define Assignment as a
	from Vars as v, Clause, Vars as v2
		where v.id=id and Clause.id=1;

define In_clause
	from Clause as c
		where c.id=2 and In_clause.clause=c;
