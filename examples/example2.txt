@entity clause: id: int;
@entity vars: id: int;
@entity in_clause: clause: int, literal: int;
@entity assignment: variable: vars;

guess at least 1 and at most 2 a with assignment(variable=v) as a 
	from vars as v
		where v, a;

define assignment as a
	from vars as v, clause, vars as v2
		where v.id=id and id=1;

define in_clause
	from clause as c
		where id=2 and clause=c.id;
