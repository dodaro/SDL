record Nurse: id: int;
record Day: id: int;
record Shift: id: int, hours: int;
record HoursLimits: min: int, max: int;
record DayLimits: shift: Shift, min: int, max: int;
record NurseLimits: shift: Shift, min: int, max: int;
record Assign: nurse: Nurse, shift: Shift, day: Day;

guess from Day, Nurse 
    exactly 1
    Assign from Shift
    where Assign.nurse == Nurse and Assign.shift == Shift and Assign.day == Day;

deny from Day, NurseLimits
    having count {
        Assign.nurse from Assign 
        where Assign.shift.id == NurseLimits.shift.id and Assign.day.id == Day.id
    } > NurseLimits.max;

deny from Day, NurseLimits
    having count {
        Assign.nurse from Assign 
        where Assign.shift.id == NurseLimits.shift.id and Assign.day.id == Day.id
    } < NurseLimits.min;

deny from Nurse, HoursLimits
    having sum {
        Shift.hours, Assign.day from Assign, Shift 
        where Assign.nurse.id == Nurse.id and Assign.shift.id == Shift.id and Assign.shift.hours == Shift.hours
    } > HoursLimits.max;

deny from Nurse, HoursLimits
    having sum {
        Shift.hours, Assign.day from Assign, Shift 
        where Assign.nurse.id == Nurse.id and Assign.shift.id == Shift.id and Assign.shift.hours == Shift.hours
    } < HoursLimits.min;

deny from Nurse
    having count {
        Assign.day from Assign 
        where Assign.nurse.id == Nurse.id and Assign.shift.id == 6
    } != 2;

deny from Nurse, Assign as a1, Assign as a2
    where a1.nurse.id == Nurse.id and a2.nurse.id == Nurse.id 
    and a2.shift.id < a1.shift.id and a1.day.id+1 == a2.day.id 
    and a2.shift.id <= 3 and a1.shift.id <= 3;

deny from Nurse, Day
    where Day.id <= 3
    having count {
        Assign.day from Assign 
        where Assign.nurse.id == Nurse.id and Assign.shift.id == 5 
        and Assign.day.id == Day.id .. Day.id+6
    } < 2;    

deny from Assign as a2, Assign as a3, not Assign as a1
    where a1.nurse.id == a2.nurse.id and a1.nurse.id == a3.nurse.id 
    and a2.shift.id == 3 and a3.shift.id == 3 and a1.shift.id == 4 
    and a3.day.id == a2.day.id + 1 
    and a1.day.id == a3.day.id + 1;

deny from Assign as a1, not Assign as a2
    where a1.nurse.id == a2.nurse.id and a1.shift.id == 4 
    and a2.shift.id == 3 and a2.day.id == a1.day.id-2;

deny from Assign as a1, not Assign as a2
    where a1.nurse.id == a2.nurse.id and a1.shift.id == 4 
    and a2.shift.id == 3 and a2.day.id == a1.day.id-1;

deny from Nurse, DayLimits
    having count{
        Assign.day from Assign 
        where Assign.shift.id == DayLimits.shift.id and Assign.nurse.id == Nurse.id
    } > DayLimits.max;

deny from Nurse, DayLimits
    having count{
        Assign.day from Assign 
        where Assign.shift.id == DayLimits.shift.id and Assign.nurse.id == Nurse.id
    } < DayLimits.min;

show Assign; 

define Day where Day.id == 1..10;
define Nurse where Nurse.id == 1..5;
define Shift where Shift.id == 1 and Shift.hours == 7;
define Shift where Shift.id == 2 and Shift.hours == 7;
define Shift where Shift.id == 3 and Shift.hours == 10;
define Shift where Shift.id == 4 and Shift.hours == 0;
define Shift where Shift.id == 5 and Shift.hours == 0;
define Shift where Shift.id == 6 and Shift.hours == 0;

define HoursLimits where HoursLimits.min == 43 and HoursLimits.max == 48;

define NurseLimits where NurseLimits.shift.id == 1 and NurseLimits.shift.hours == 7 and NurseLimits.min == 1 and NurseLimits.max == 2;
define NurseLimits where NurseLimits.shift.id == 2 and NurseLimits.shift.hours == 7 and NurseLimits.min == 1 and NurseLimits.max == 2;
define NurseLimits where NurseLimits.shift.id == 3 and NurseLimits.shift.hours == 10 and NurseLimits.min == 0 and NurseLimits.max == 1;

define DayLimits where DayLimits.shift.id == 1 and DayLimits.shift.hours == 7 and DayLimits.min == 2 and DayLimits.max == 3;
define DayLimits where DayLimits.shift.id == 2 and DayLimits.shift.hours == 7 and DayLimits.min == 2 and DayLimits.max == 3;
define DayLimits where DayLimits.shift.id == 3 and DayLimits.shift.hours == 10 and DayLimits.min == 1 and DayLimits.max == 2;
