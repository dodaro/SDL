record Color: value: any;
record Node: value: int;
record Edge: node1: Node, node2: Node;
record Assign: node: Node, color: Color;


define Assign as a 
	where a.node.value==1 and a.color.value=="blue";
	
show Color, Node, Edge, Assign;

define Color as c
	where c.value=="red";

define Node as n 
	where n.value==1;
	

define Node as n
	from Assign as a
		where n.value==a.node.value
	having max {a2.node.value from Assign as a2 where a2.node.value==1} >= n.value and
	sum {a4.node.value from Assign as a4} ==1;

define Color as c
    from Assign, Node as n
	where Assign.color==c and n.value==5;
