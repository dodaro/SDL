/* Represents a patient to be scheduled */
record Registration: 
    id: int, 
    priority: int, 
    surgery_duration: int, 
    length_of_stay: int, 
    specialty: int, 
    icu_days: int, 
    advance_admission: int;

/* Represents a Medical Surgical Session (an operating room time slot) */
record Mss: 
    room: int, 
    sequence: int, 
    specialty: int, 
    day: int;

/* Represents the maximum available surgery time for a specific room and sequence */
record Duration: 
    max_minutes: int, 
    room: int, 
    sequence: int;

/* Represents the daily bed capacity for standard wards and ICU */
record Beds: 
    specialty: int, 
    available_beds: int, 
    day: int;

/* Represents the scheduling horizon (days) */
record Day: 
    id: int;

/* The decision variable: maps a patient to a medical surgical session */
record Assign: 
    registration: Registration, 
    mss: Mss;

/* Derived fact: a patient occupying a standard hospital bed on a specific day */
record Stay: 
    registration: Registration, 
    day: Day;

/* Derived fact: a patient occupying an Intensive Care Unit (ICU) bed on a specific day */
record StayICU: 
    registration: Registration, 
    day: Day;

/* Generates the possible assignments of patients to surgical sessions.
   A patient can only be assigned to a session if their required medical specialty 
   matches the specialty of the session.
*/
guess from Registration, Mss
    Assign
    where Assign.registration == Registration 
      and Assign.mss == Mss 
      and Registration.specialty == Mss.specialty;

/* A patient cannot be scheduled for two different sequences (time slots) in the same operating room */
deny from Assign as a1, Assign as a2
    where a1.registration == a2.registration 
      and a1.mss.room == a2.mss.room 
      and a1.mss.sequence != a2.mss.sequence;

/* A patient cannot be scheduled in two different operating rooms during the exact same sequence (time slot) */
deny from Assign as a1, Assign as a2
    where a1.registration == a2.registration 
      and a1.mss.sequence == a2.mss.sequence 
      and a1.mss.room != a2.mss.room;

/* The total surgery duration of all patients assigned to a specific room and sequence cannot exceed its maximum duration */
deny from Duration
    having sum {
        Assign.registration.surgery_duration, Assign.registration from Assign
        where Assign.mss.room == Duration.room 
          and Assign.mss.sequence == Duration.sequence
    } > Duration.max_minutes;

/* Calculates pre-surgery ward stay based on admission days in advance */
define Stay
    from Assign, Day
    where Assign.registration.advance_admission > 0
      and Day.id >= Assign.mss.day - Assign.registration.advance_admission
      and Day.id < Assign.mss.day
      and Stay.registration == Assign.registration
      and Stay.day == Day;

/* Calculates post-ICU ward stay based on total length of stay and ICU days */
define Stay
    from Assign, Day
    where Assign.registration.length_of_stay > Assign.registration.icu_days
      and Day.id >= Assign.mss.day + Assign.registration.icu_days
      and Day.id < Assign.mss.day + Assign.registration.length_of_stay
      and Stay.registration == Assign.registration
      and Stay.day == Day;

/* Calculates Intensive Care Unit (ICU) stay directly after surgery */
define StayICU
    from Assign, Day
    where Assign.registration.icu_days > 0
      and Day.id >= Assign.mss.day
      and Day.id < Assign.mss.day + Assign.registration.icu_days
      and StayICU.registration == Assign.registration
      and StayICU.day == Day;

/* Standard ward bed capacity: Staying patients for a specialty on a specific day cannot exceed available beds */
deny from Beds
    where Beds.specialty > 0
    having count {
        Stay.registration from Stay
        where Stay.registration.specialty == Beds.specialty 
          and Stay.day.id == Beds.day
    } > Beds.available_beds;

/* ICU bed capacity (specialty == 0): Total patients in the ICU on a specific day cannot exceed available ICU beds */
deny from Beds
    where Beds.specialty == 0
    having count {
        StayICU.registration from StayICU
        where StayICU.day.id == Beds.day
    } > Beds.available_beds;

/* Priority 1 patients are mandatory. They must be assigned at least once. */
deny from Registration
    where Registration.priority == 1
    having count {
        Assign.registration.id from Assign where Assign.registration.id == Registration.id
    } == 0;

/* Priority 2 patients are highly important. If unassigned, incur a heavy penalty at level 3. */
deny from Registration
    where Registration.priority == 2
    having count {
        Assign.registration.id from Assign where Assign.registration.id == Registration.id
    } == 0
    or pay 1 at 3;

/* Priority 3 patients are less urgent. If unassigned, incur a lighter penalty at level 2. */
deny from Registration
    where Registration.priority == 3
    having count {
        Assign.registration.id from Assign where Assign.registration.id == Registration.id
    } == 0
    or pay 1 at 2;

deny from Registration
    having count { Assign from Assign
                   where Assign.registration == Registration } > 1;

/* Tell the solver to output the final assignments */
show Assign;
