record Node: id: int;
record Edge: first: Node, second: Node;
record Color: value: str;
record Assign: node: Node, color: Color;


define Node as n
	where n.id==1;	
define Node as n
	where n.id==2;
define Node as n
	where n.id==3;
define Node as n
	where n.id==4;
define Node as n
	where n.id==5;
guess from Node as n
    exactly 1 
	Assign as a
        	from Color
            		where a.node==n and a.color==Color;
            
deny from Assign as a1, Assign as a2, Edge
    where  a1.node.id != a2.node.id and a1.color == a2.color and Edge.first == a1.node and Edge.second==a2.node;

define Edge as e
    from Node as n1, Node as n2
        where e.first == n1 and e.second== n2 and n1.id==1 and n2.id==2;
